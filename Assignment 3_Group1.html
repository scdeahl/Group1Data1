# Define a function to get the author's name from a URL
def get_author(url):
    # If the URL is missing or not a string, return 'unknown'
    if pd.isna(url) or not isinstance(url, str):
        return 'unknown'
    
    # Try to get the author's name from the URL
    try:
        # Split the URL into parts using '/' and take the last part
        last_part = url.split('/')[-1]
        # Split the last part by '.' and get the second part
        name = last_part.split('.')[1]
        # Replace underscores with spaces to make the name look nice
        name = name.replace('_', ' ')
    except IndexError:
        # If there is an error, return 'unknown'
        return 'unknown'
    
    return name

# Use the function to make a new 'author' column from 'author_url'
