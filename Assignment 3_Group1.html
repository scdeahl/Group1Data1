#writing a function, that clears the url from everything except the name of the author
def get_author(url):

    name = url.split('.')[-1]   # taking the part after the last dot
    name = name.split('_')      # spliting the text into a list, divided by '_'
    name = ' '.join(name)       # forming a string from the items of the list
    return name

#creating a new column with the function
gr_subset['author']=gr_subset.author_url.map(get_author)
